cmake_minimum_required(VERSION 3.23)
project(liquid)
set(CMAKE_CXX_STANDARD 11)
add_executable(output.exe src/main.cpp)
include_directories(liquid)

set(CURSES_NEED_NCURSES TRUE)
set(CURSES_NEED_WIDE TRUE)
find_package(Curses)
target_include_directories(output.exe PUBLIC ${CURSES_INCLUDE_DIR})
target_link_libraries(output.exe ${CURSES_LIBRARIES})
